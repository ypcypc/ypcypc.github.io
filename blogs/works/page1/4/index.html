#include<cstdio>
#include<windows.h>
#include<iostream>
#include<ctime>
using namespace std;
POINT Search_Box,Close_Position,Search_Button;
int n,tot,Now_Length=2;
double Wait_Time;
bool If_Need_Search_Button=1;
inline void click(POINT xx){
    SetCursorPos(xx.x,xx.y);
    mouse_event(MOUSEEVENTF_LEFTDOWN,0,0,0,0);
    Sleep(5);
    mouse_event(MOUSEEVENTF_LEFTUP,0,0,0,0);
    Sleep(5);
}
inline void Carry_Button(char x){
    keybd_event((int)x,0,0,0);
    Sleep(5);
    keybd_event((int)x,0,KEYEVENTF_KEYUP,0);
    Sleep(5);
}
inline void Input_Information(string Information){
    if (Information.length()>Now_Length||++tot>n)return;
    click(Search_Box);
    Sleep(5);
    for (int i=0;i<=Information.length();i++)Carry_Button(VK_BACK);
    for (int i=0;i<Information.length();i++)Carry_Button(Information[i]);
    if (If_Need_Search_Button)click(Search_Button);else Carry_Button(VK_RETURN);
    Sleep(Wait_Time);
    click(Close_Position);
    for (char i='9';i>='1';i--)Input_Information(Information+i);
    for (char i='A';i<='Z';i++)Input_Information(Information+i);
}
inline void Read_Config(){
    printf("请在此输入您想保持浏览器活跃几次：");scanf("%d",&n);
    printf("\n请在此输入每一次浏览器搜索页面打开的最大时间（单位：秒）");
    printf("\n此时间依据您的浏览器快慢而定，普通网速建议设置成1秒：");scanf("%lf",&Wait_Time);
    Wait_Time*=1000;
    system("title 下面进入确定您浏览器各个位置的时间，请根据向导完成操作");
    while (1){
        system("cls");
        printf("\n请将您的鼠标移动到您的浏览器的搜索框的位置，已移动到请按1键：");
        int Check_If_Complete;scanf("%d",&Check_If_Complete);
        if (Check_If_Complete==1){GetCursorPos(&Search_Box);break;}
    }
    while (1){
        system("cls");
        printf("\n请将您的鼠标移动到您的浏览器的搜索按钮的位置，已移动到请按1键\n");
        printf("注意，如果这里你习惯用Enter键搜索请输入2：");
        int Check_If_Complete;scanf("%d",&Check_If_Complete);
        if (Check_If_Complete==1){GetCursorPos(&Search_Button);break;}
                            else if (Check_If_Complete==2){If_Need_Search_Button=0;break;}
    }
    while (1){
        system("cls");
        printf("\n请将您的鼠标移动到您的浏览器的关闭浏览页面（并非关闭浏览器）的位置，已移动到请按1键：");
        int Check_If_Complete;scanf("%d",&Check_If_Complete);
        if (Check_If_Complete==1){GetCursorPos(&Close_Position);break;}
    }
}
int main(){
    Read_Config();
    while (tot<n){
        for (char i='A';i<='Z';i++){
            string Start_String="";
            Input_Information(Start_String+i);
        } 
        Now_Length++;
    }
    printf("运行完成！一共使浏览器保持活跃%0.3lf秒，搜索了%d个信息！",clock()/1000.0,n);
    system("pause");
    return 0;
}
